      function ensureAudio() {
      if (!audioContext) {
      audioContext = new (window.AudioContext || window.webkitAudioContext)();

      audioContext.resume();

      function playGoalHorn() {
      ensureAudio();
      if (!audioContext) return;
      const now = audioContext.currentTime;
      const gain = audioContext.createGain();
      gain.connect(audioContext.destination);
      gain.gain.setValueAtTime(0.0001, now);
      gain.gain.exponentialRampToValueAtTime(0.22, now + 0.02);
      gain.gain.exponentialRampToValueAtTime(0.0001, now + 0.6);
      const oscA = audioContext.createOscillator();
      oscA.type = "sawtooth";
      oscA.frequency.setValueAtTime(220, now);
      oscA.frequency.linearRampToValueAtTime(280, now + 0.3);
      const oscB = audioContext.createOscillator();
      oscB.type = "square";
      oscB.frequency.setValueAtTime(330, now);
      oscB.frequency.linearRampToValueAtTime(410, now + 0.3);
      oscA.connect(gain);
      oscB.connect(gain);
      oscA.start(now);
      oscB.start(now + 0.03);
      oscA.stop(now + 0.62);
      oscB.stop(now + 0.56);
      }
      function setPaused(nextPaused) {

      function movePaddleKeyboard(paddle, left, right, up, down, minX, maxX) {
      paddle.prevX = paddle.x;
      paddle.prevY = paddle.y;
      let dx = 0;
      let dy = 0;
      if (left) dx -= 1;
      if (right) dx += 1;
      if (up) dy -= 1;
      if (down) dy += 1;
      const len = Math.hypot(dx, dy);
      if (len > 0) {
      dx = (dx / len) * paddle.maxSpeed;
      dy = (dy / len) * paddle.maxSpeed;
      }
      paddle.x = clamp(paddle.x + dx, minX, maxX);
      paddle.y = clamp(paddle.y + dy, paddle.r + 10, H - paddle.r - 10);
      }
      function movePlayer() {
      movePaddleKeyboard(
      player,
      keys.a,
      keys.d,
      keys.w,
      keys.s,
      25,
      config.playerMaxX - player.r
      );
      }

      movePaddleKeyboard(
      player2,
      keys.left,
      keys.right,
      keys.up,
      keys.down,
      config.aiMinX + player2.r,
      W - player2.r - 25
      );
      }
      function getAiParams() {
      return AI_BY_DIFFICULTY[state.difficulty] || AI_BY_DIFFICULTY.medium;
      }
      function moveAI() {
      player2.prevX = player2.x;
      player2.prevY = player2.y;
      const { speed: cap, lead } = getAiParams();
      let targetX = puck.x + lead;
      let targetY = puck.y;
      if (puck.vx < 0 && puck.x < CENTER_X + 90) {
      targetX = W * 0.75;
      targetY = H * 0.5;
      }
      targetX = clamp(targetX, config.aiMinX + player2.r, W - player2.r - 25);
      targetY = clamp(targetY, player2.r + 10, H - player2.r - 10);
      const dx = targetX - player2.x;
      const dy = targetY - player2.y;
      const dist = Math.hypot(dx, dy);
      if (dist > cap) {
      const t = cap / dist;
      player2.x += dx * t;
      player2.y += dy * t;
      } else {
      player2.x = targetX;
      player2.y = targetY;
      }
      }
      function separateBodies(a, b) {
      const dx = b.x - a.x;
      const dy = b.y - a.y;
      const dist = Math.hypot(dx, dy) || 0.0001;
      const overlap = a.r + b.r - dist;
      if (overlap > 0) {
      const nx = dx / dist;
      const ny = dy / dist;
      b.x += nx * overlap;
      b.y += ny * overlap;
      }
      }
      // Puck speed changes based on impact strength and direction.
      function handlePaddleCollision(paddle) {
      const dx = puck.x - paddle.x;
      const dy = puck.y - paddle.y;
      const dist = Math.hypot(dx, dy);
      const minDist = puck.r + paddle.r;
      if (dist >= minDist || dist === 0) return;
      separateBodies(paddle, puck);
      const nx = (puck.x - paddle.x) / (Math.hypot(puck.x - paddle.x, puck.y - paddle.y) || 1);
      const ny = (puck.y - paddle.y) / (Math.hypot(puck.x - paddle.x, puck.y - paddle.y) || 1);
      const paddleVx = paddle.x - paddle.prevX;
      const paddleVy = paddle.y - paddle.prevY;
      const relVx = puck.vx - paddleVx;
      const relVy = puck.vy - paddleVy;
      const speedIntoNormal = relVx * nx + relVy * ny;
      if (speedIntoNormal < 0) {
      const bounce = 1.05;
      puck.vx -= (1 + bounce) * speedIntoNormal * nx;
      puck.vy -= (1 + bounce) * speedIntoNormal * ny;
      }
      const hitStrength = Math.hypot(paddleVx, paddleVy);
      const boostActive =
      (paddle === player && state.boostP1) ||
      (paddle === player2 && state.mode === "multi" && state.boostP2);
      const boostMultiplier = boostActive ? 1.8 : 1;
      const extraSpeed = clamp(hitStrength * 0.85 * boostMultiplier, 0, 10.2);
      const currentSpeed = Math.hypot(puck.vx, puck.vy);
      const desiredSpeed = clamp(currentSpeed + extraSpeed, config.minPuckSpeedAfterHit, config.maxPuckSpeed);
      if (currentSpeed > 0.01) {
      const scale = desiredSpeed / currentSpeed;
      puck.vx *= scale;
      puck.vy *= scale;
      } else {
      puck.vx = nx * desiredSpeed;
      puck.vy = ny * desiredSpeed;
      }
      }
      function updatePuck() {
      puck.x += puck.vx;
      puck.y += puck.vy;
      const goalTop = (H - config.goalHeight) / 2;
      const goalBottom = goalTop + config.goalHeight;
      const inGoalBand = puck.y > goalTop && puck.y < goalBottom;
      if (puck.y <= puck.r + 8) {
      puck.y = puck.r + 8;
      puck.vy *= -config.wallBounce;
      }
      if (puck.y >= H - puck.r - 8) {
      puck.y = H - puck.r - 8;
      puck.vy *= -config.wallBounce;
      }
      if (puck.x <= puck.r + 8 && !inGoalBand) {
      puck.x = puck.r + 8;
      puck.vx *= -config.wallBounce;
      }
      if (puck.x >= W - puck.r - 8 && !inGoalBand) {
      puck.x = W - puck.r - 8;
      puck.vx *= -config.wallBounce;
      }
      if (puck.x < 0) {
      state.player2Score += 1;
      player2ScoreEl.textContent = String(state.player2Score);
      playGoalHorn();
      if (state.player2Score >= WIN_SCORE) {
      endMatch(false);
      } else {
      resetPuck(1);
      }
      } else if (puck.x > W) {
      state.playerScore += 1;
      playerScoreEl.textContent = String(state.playerScore);
      playGoalHorn();
      if (state.playerScore >= WIN_SCORE) {
      endMatch(true);
      } else {
      resetPuck(-1);
      }
      }
      puck.vx *= config.friction;
      puck.vy *= config.friction;
      const speed = Math.hypot(puck.vx, puck.vy);
      if (speed > config.maxPuckSpeed) {
      const s = config.maxPuckSpeed / speed;
      puck.vx *= s;
      puck.vy *= s;
      }
      if (speed < 0.28) {
      puck.vx += (Math.random() - 0.5) * 0.14;
      puck.vy += (Math.random() - 0.5) * 0.14;
      }
      }

      }

      tag === "INPUT" ||
      tag === "SELECT" ||
      tag === "TEXTAREA" ||
      tag === "A"
      );
      }

      drawRink();

      moveAI();

      handlePaddleCollision(player2);
      updatePuck();
      }
      drawObjects();
      if (state.isPaused && !state.gameOver) drawPauseOverlay();
      }
      requestAnimationFrame(tick);
      }
      function applyMovementKey(event, down) {
      const code = event.code;
      const key = event.key;
      if (code === "KeyW" || key === "w" || key === "W") {
      keys.w = down;
      return true;
      }
      if (code === "KeyA" || key === "a" || key === "A") {
      keys.a = down;
      return true;
      }
      if (code === "KeyS" || key === "s" || key === "S") {
      keys.s = down;
      return true;
      }
      if (code === "KeyD" || key === "d" || key === "D") {
      keys.d = down;
      return true;
      }
      if (state.mode === "multi") {
      if (code === "ArrowUp" || key === "ArrowUp") {
      keys.up = down;
      return true;
      }
      if (code === "ArrowDown" || key === "ArrowDown") {
      keys.down = down;
      return true;
      }
      if (code === "ArrowLeft" || key === "ArrowLeft") {
      keys.left = down;
      return true;
      }
      if (code === "ArrowRight" || key === "ArrowRight") {
      keys.right = down;
      return true;
      }
      }
      return false;
      }
      function onKeyDown(event) {
      if (state.screen !== "playing" || state.gameOver) {
      return;
      }
      if (event.repeat && applyMovementKey(event, true)) {
      event.preventDefault();
      return;
      }
      if (event.repeat) {
      if (event.code === "Space") {
      event.preventDefault();
      }
      return;
      }
      if (event.code === "Space") {
      if (!isFormOrButtonFocused()) {
      state.boostP1 = true;
      event.preventDefault();
      ensureAudio();
      }
      return;
      }
      if (event.code === "Enter" && state.mode === "multi") {
      if (!isFormOrButtonFocused()) {
      state.boostP2 = true;
      event.preventDefault();
      ensureAudio();
      }
      return;
      }
      if (event.code === "KeyR" || event.key === "r" || event.key === "R") {
      beginRound();
      event.preventDefault();
      return;
      }
      if (event.code === "KeyP" || event.key === "p" || event.key === "P") {
      const nextPaused = !state.isPaused;
      setPaused(nextPaused);
      event.preventDefault();
      if (!nextPaused) {
      requestAnimationFrame(() => focusGameInput());
      }
      return;
      }
      if (applyMovementKey(event, true)) {
      event.preventDefault();
      }
      }
      function onKeyUp(event) {
      if (state.screen !== "playing") {
      return;
      }
      if (event.code === "Space") {
      state.boostP1 = false;
      event.preventDefault();
      return;
      }
      if (event.code === "Enter") {
      state.boostP2 = false;
      if (state.mode === "multi") event.preventDefault();
      return;
      }
      if (applyMovementKey(event, false)) {
      if (event.code.startsWith("Arrow") && state.mode === "multi") {
      event.preventDefault();
      }
      }
      }
